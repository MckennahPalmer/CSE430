```
saveInternalData(query) : tuple

    queryParts = query.split(",")
    action = queryParts[0]
    element = queryParts[1]
    data = queryParts[2] 

		 
  # OPTION 1:
  if element == "schedules"
    result = schedulingDatabaseManager(action, data) # see 1.3.2 Scheduling Database Data Request, maybe this is not needed.
    return (result.status, result.data)
	
  # OPTION2, if we don't use schedulingDatabaseManager:
  mydb = mysql.connector.connect(
    host="enrollease",
    user="username",
    password="password",
    database="scheduling-database"
  )
    
  mycursor = mydb.cursor()
  beforeRowId = mycursor.lastrowid
  
  if action == "INSERT"
  	if element == "schedules"    

      sql = "INSERT INTO schedules"
      val = data
      mycursor.execute(sql, val)
      mydb.commit()
  
  if beforeRowID != mycursor.lastrowid
    return ("SUCCESS", mycursor.lastrowid)
  else
    return ("ERROR", data)
```
